  <div class="file-wrapper">

    <input type="file" id="fileAttach" name="fileAttach" title="파일업로드" class="file-input" />

<script>
$(document).ready(function () {
  fileAttach();
});
function fileAttach() {
  $(document).on('change', 'input.file-input', function () {
    var $tg = $(this);
    var $name = $tg.val().split('\\').pop();
    $tg.closest('.input-cover').addClass('active').find('.result-file-input').val($name);
  });
}
</script>
